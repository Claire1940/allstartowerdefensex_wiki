<Section icon={Shield} title="Survival Steps">

<Steps>
  <Step num="1" title="Early Game Setup">
    Place your economy units immediately. Focus on generating enough cash to support high-tier unit summons before wave 10.
  </Step>
  <Step num="2" title="Mid-Game Stabilization">

  </Step>

  </Step>
</Steps>
